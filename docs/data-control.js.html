var preloader = require('./preloader');
var radio = require('./radio');

/**
 * Data Control loads assets and parses data objects
 * @package Core
 * @module  dataControl
 */
module.exports = {
    /**
     * Loads assets and parses data objects
     *
     * @param  {object} data - the data object that describes what will be loaded into the game
     * @param  {object} options - options.types - the types of entity types needed for the data being loaded
     */
    parse: function(data, options) {
        this._data = data;
        this._types = options.types;
        
        radio.tuneIn(document, 'preloader/assetsloaded', this._onAssetsLoaded, this);

        // preload assets
        preloader.load(data.assets);
    },

    /**
     * @private
     */
    _onAssetsLoaded: function() {
        var data = this._data,
            parsed = {
                entities: {}
            },
            entity,
            prop,
            len,
            i;

        radio.tuneOut(document, 'preloader/assetsloaded', this._onAssetsLoaded);

        len = data.entities.length;
        for (i = 0; i < len; i += 1) {
            entity = data.entities[i];

            parsed.entities[entity.name] = new this._types[entity.type]();

            if (entity.config.hasOwnProperty('src')) {
                parsed.entities[entity.name].setImage(entity.config.src);
            }

            if (entity.config.hasOwnProperty('x')) {
                parsed.entities[entity.name].setX(entity.config.x);
            }

            if (entity.config.hasOwnProperty('y')) {
                parsed.entities[entity.name].setY(entity.config.y);
            }
        }

        radio.broadcast('dataparsed', {
            data: parsed
        });
    }
};
