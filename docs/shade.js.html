var config = require('./config');
var radio = require('./radio.js');
var proto = require('./proto.js');

/**
 * A base class for movable objects
 * @package Core
 * @class   Shade
 */
module.exports = proto.create({
    /** 
     * @member {string} Shade.prototype.name - the unique name necessary for proto's inheritance
     */
    name: 'shade',

    init: function() {
        this._x = 0;
        this._y = 0;
        this._vx = 0;
        this._vy = 0;
        this._screenX = 0;
        this._screenY = 0;
        this._width = 0;
        this._height = 0;
        this._halfWidth = 0;
        this._halfHeight = 0;
    },

    _update: function() {
        // apply velocity
        this._x += this._vx;
        this._y += this._vy;
        // set render x/y
        this._screenX = this._x;
        this._screenY = this._y;
    },

    _parseVal: function(n, key) {
        if (typeof n === 'number') {
            this[key] = n;
        // matches '+=n' or '-=n' or '+=n.n'
        } else if (/^\+|\-\=[0-9\.]+$/.test(n)) {
            this[key] += parseFloat(n.match(/[0-9\.]+/), 10);
        }
    },

    /**
     * Sets or adds to the object's x coordinate
     * @method Shade.prototype.setX
     * @param {number|string} x - a number sets the value, a string value of '+=x' or '-=x' adds or subtracts the current value resprectively
     * @returns this
     */
    setX: function(x) {
        this._parseVal(x, '_x');
        return this;
    },

    /**
     * Sets or adds to the object's y coordinate
     * @method Shade.prototype.setY
     * @param {number|string} y - a number sets the value, a string value of '+=y' or '-=y' adds or subtracts the current value resprectively
     * @returns this
     */
    setY: function(y) {
        this._parseVal(y, '_y');
        return this;
    },

    /**
     * Sets or adds to the object's vx coordinate
     * @method Shade.prototype.setVx
     * @param {number|string} vx - a number sets the value, a string value of '+=vx' or '-=vx' adds or subtracts the current value resprectively
     * @returns this
     */
    setVx: function(vx) {
        this._parseVal(vx, '_vx');
        return this;
    },

    /**
     * Sets or adds to the object's vy coordinate
     * @method Shade.prototype.setVy
     * @param {number|string} vy - a number sets the value, a string value of '+=vy' or '-=vy' adds or subtracts the current value resprectively
     * @returns this
     */
    setVy: function(vy) {
        this._parseVal(vy, '_vy');
        return this;
    },

    /**
     * Sets the object's width value
     * @method Shade.prototype.setwidth
     * @param {number} width - the new width value
     * @returns this
     */
    setWidth: function(width) {
        this._parseVal(width, '_width');
        return this;
    },

    /**
     * Sets the object's height value
     * @method Shade.prototype.setHeight
     * @param {number} height - the new height value
     * @returns this
     */
    setHeight: function(height) {
        this._parseVal(height, '_height');
        return this;
    },

    /**
     * @method Shade.prototype.getX
     * @returns {number} x
     */
    getX: function() {
        return this._x;
    },

    /**
     * @method Shade.prototype.getY
     * @returns {number} y
     */
    getY: function() {
        return this._y;
    },

    /**
     * @method Shade.prototype.getVx
     * @returns {number} vx
     */
    getVx: function() {
        return this._vx;
    },

    /**
     * @method Shade.prototype.getVy
     * @returns {number} vy
     */
    getVy: function() {
        return this._vy;
    },

    /**
     * @method Shade.prototype.getHeight
     * @returns {number} height
     */
    getHeight: function() {
        return this._height;
    },

    /**
     * @method Shade.prototype.getWidth
     * @returns {number} width
     */
    getWidth: function() {
        return this._width;
    }
});
