var config = require('./config');
var Shade = require('./shade');

/**
 * controls what to show on the canvas
 * @package Core
 * @class   Camera
 * @extends Shade
 */
module.exports = Shade.extend({
    /** 
     * @member {string} Camera.prototype.name - the unique name necessary for proto's inheritance
     */
    name: 'camera',

    init: function() {
        this.$shade.init.call(this);

        this._width = config.width;
        this._height = config.height;

        this._zoom = 1;
    },

    /**
     * returns the zoom property
     * @method  Camera.prototype.getZoom
     * @returns canvas
     */
    getZoom: function() {
        return this._zoom;
    },

    /**
     * sets the zoom property
     * @method  Camera.prototype.setZoom
     * @params  {number} zoom
     * @returns Camera
     */
    setZoom: function(zoom) {
        this._zoom = zoom;
        return this;
    }
});
