var Event = require('./event');

/**
 * An observable object that emits events when changed
 * @package Core
 * @class  Overt
 * @extends Event
 */
module.exports = Event.extend({
    /** 
     * @member {string} Overt.prototype.name - the unique name necessary for proto's inheritance
     */
    name: 'overt',

    init: function() {
        this.data = {};
        this.$event.init.call(this);
    },

    /**
     * @method Overt.prototype.set
     * @param {object} pairs - key/val pairs to set
     * @param {object} options - 
     */
    set: function(obj, options) {
        var old = {},
            prop;

        for (prop in obj) {
            old[prop] = this.data[prop];
            this.data[prop] = obj[prop];
            this.announce('change', {
                'key': prop,
                'oldValue': old[prop],
                'newValue': obj[prop]
            });
        }
    },

    /**
     * @method Overt.prototype.get
     * @param {string} prop - the property
     * @returns {any} property
     */
    get: function(prop) {
        return this.data[prop];
    }
});
