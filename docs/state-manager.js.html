var radio = require('./radio');

/**
 * State Manager registers state constructors and stores their prototypes for loading
 * @package Core
 * @module  stateManager
 */
module.exports = {
    _states: {},
    _currentState: null,

    /**
     * Adds a new state prototype to the pool
     * @param {constructor} State - the State constrctor
     * @param {object} [options] - options to pass to the constructor
     */
    add: function(State, options) {
        var state = new State(options);
        this._states[state.name] = state;
    },

    /**
     * Get a state based on its name
     * @param {string} name - the same name property passed in proto.create or State.extend
     * @return {object} state - the state prototype
     */
    get: function(name) {
        return this._states[name];
    },

    /**
     * After parsing the appropriate data, destroys old state and sets up new one
     * @param {string} name - the same name property passed in proto.create or State.extend
     * @return {object} state - the state prototype
     */
    load: function(data, state) {
        if (this._currentState) {
            radio.tuneOut('newframe', this._currentState.update);
        } else {
            this._currentState = this._states[state];
        }

        this._currentState.init(data);

        if (typeof this._currentState.update === 'function') {
            // pass current state for context
            radio.tuneIn('newframe', this._currentState.update, this._currentState);
        }
    }
};
