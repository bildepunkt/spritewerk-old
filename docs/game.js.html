var config       = require('./config');
var domControl   = require('./dom-control');
var draw         = require('./draw');
var engine       = require('./engine');
var shade        = require('./shade');
var radio        = require('./radio.js');
var stateManager = require('./state-manager');
var entityManager = require('./entity-manager');
var dataControl  = require('./data-control');

/**
 * Game loads data and state functionality
 * @package Core
 * @module  game
 */
module.exports = {
    /**
     * calls loadDataAndState on window load
     *
     * @param  {object} data - a data object
     * @param  {string} state - the state name
     * @param  {object} options - key/val pair of options
     */
    init: function(data, state, options) {
        this._state = state;
        this._data = data;
        this._options = options;

        radio.tuneIn(window, 'load', this._onWindowLoad, this);
    },

    /**
     * @private
     */
    _onWindowLoad: function() {
        radio.tuneOut(window, 'load');

        domControl.init();
        entityManager.init();
        draw.init();
        engine.init();

        this.loadDataAndState(this._data, this._state, this._options);
    },

    /**
     * @private
     */
    _onDataParsed: function(e) {
        radio.tuneOut('dataparsed', this._onDataParsed);
        stateManager.load(e.detail.data, this._state);
    },

    /**
     * loads data and state functionality to run the game
     *
     * @param  {object} data - a data object
     * @param  {string} state - the state name
     * @param  {object} options - key/val pair of options
     */
    loadDataAndState: function(data, state, options) {
        this.data = data;
        this._state = state;
        this._options = options;

        radio.tuneIn('dataparsed', this._onDataParsed, this);

        dataControl.parse(data, options);
    }
};
